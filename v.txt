 $files = Import-Csv "C:\future_files.csv"
  $now = Get-Date
 
  foreach ($row in $files) {
      try {
          $file = Get-Item -Path $row.FullName -ErrorAction Stop
 
          # Check if any of the three main timestamps are in the future
          if ($file.CreationTime -gt $now -or $file.LastWriteTime -gt $now -or $file.LastAccessTime -gt $now) {
 
              # 1. Take ownership and reset permissions to ensure access
              takeown /f $file.FullName
              icacls "$($file.FullName)" /grant "${env:USERNAME}:F"
 
              # 2. Reset the dates to the current time
              # You can also set a specific past date if preferred
              $file.CreationTime = $now
              $file.LastWriteTime = $now
              $file.LastAccessTime = $now
 
              Write-Host "Reset future dates for: $($file.Name)" -ForegroundColor Cyan
          }
      } catch {
          Write-Warning "Could not process $($row.FullName). Error: $($_.Exception.Message)"
      }
  }
